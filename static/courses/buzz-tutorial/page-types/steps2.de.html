In der course.json-Datei muss eine Seite mit Unterseiten definiert werden. Folgende Seitentypen stehen dabei zur Verfügung:

steps-large
Ablaufschritte mit einer Maximalbreite von 1280 px
steps-medium
Ablaufschritte mit einer Maximalbreite von 960 px
steps-small
Ablaufschritte mit einer Maximalbreite von 480 px
{
    "language": {
        "": {
            "name": "Buzz Tutorial",
            "pages": {
                ...
                "page-types/tabs/": {
                    "name": "Schritt-Für-Schritt-Abläufe",
                    "type": "steps-large",
                    "pages": {
                        "1": {
                            "pos": 1,
                            "name": "So sieht es aus",
                            "file": "page-types/steps1.de.html"
                        },
                        "2": {
                            "pos": 2,
                            "name": "Beispiel für course.json",
                            "file": "page-types/steps2.de.html"
                        },
                        "3": {
                            "pos": 3,
                            "name": "Inhalt der HTML-Dateien",
                            "file": "page-types/steps3.de.html"
                        }
                    }
                }
            }
        }
    }
}
